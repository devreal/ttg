528  // Returns true if no null pointers encountered (i.e., if all
529  // encountered terminals/operations are connected)
530  bool traverse(OpBase *op) {
531  if (!op) {
532  std::cout << "ttg::Traverse: got a null op!\n";
533  return false;
534  }
535 
536  if (visited(op)) return true;
537 
538  bool status = true;
539 
540  opfunc(op);
541 
542  int count = 0;
543  for (auto in : op->get_inputs()) {
544  if (!in) {
545  std::cout << "ttg::Traverse: got a null in!\n";
546  status = false;
547  } else {
548  infunc(in);
549  if (!in->is_connected()) {
550  std::cout << "ttg::Traverse: " << op->get_name() << " input terminal #" << count << " " << in->get_name() << " is not connected\n";
551  status = false;
552  }
553  }
554  count++;
555  }
556 
557  count = 0;
558  for (auto out : op->get_outputs()) {
559  if (!out) {
560  std::cout << "ttg::Traverse: got a null out!\n";
561  status = false;
562  } else {
563  outfunc(out);
564  if (!out->is_connected()) {
565  std::cout << "ttg::Traverse: " << op->get_name() << " output terminal #" << count << " " << out->get_name() << " is not connected\n";
566  status = false;
567  }
568  }
569  count++;
570  }
571 
572  for (auto out : op->get_outputs()) {
573  if (out) {
574  for (auto successor : out->get_connections()) {
575  if (!successor) {
576  std::cout << "ttg::Traverse: got a null successor!\n";
577  status = false;
578  } else {
579  status = status && traverse(successor->get_op());
580  }
581  }
582  }
583  }
584 
585  return status;
586  }

1172  // connect a single output terminal with a single
